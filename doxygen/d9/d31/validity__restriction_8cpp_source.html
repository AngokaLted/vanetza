140 size_t deserialize(InputArchive& ar, ValidityRestriction& restriction)
141 {
142  size_t size = 0;
143  ValidityRestrictionType type;
144  deserialize(ar, type);
145  size += sizeof(ValidityRestrictionType);
146  switch (type) {
147  case ValidityRestrictionType::Time_End: {
148  EndValidity end;
149  deserialize(ar, end);
150  size += sizeof(end);
151  restriction = end;
152  break;
153  }
154  case ValidityRestrictionType::Time_Start_And_Duration: {
155  StartAndDurationValidity validity;
156  deserialize(ar, validity.start_validity);
157  uint16_t duration;
158  deserialize(ar, duration);
159  validity.duration = Duration(duration);
160  size += get_size(validity);
161  restriction = validity;
162  break;
163  }
164  case ValidityRestrictionType::Time_Start_And_End: {
165  StartAndEndValidity validity;
166  deserialize(ar, validity.start_validity);
167  deserialize(ar, validity.end_validity);
168  restriction = validity;
169  size += get_size(validity);
170  break;
171  }
172  case ValidityRestrictionType::Region: {
173  GeographicRegion region;
174  size += deserialize(ar, region);
175  restriction = region;
176  break;
177  }
178  default:
179  throw deserialization_error("Unknown ValidityRestrictionType");
180  }
181  return size;
182 }
