204 {
205  DataConfirm result;
206  result ^= validate_data_request(request, m_mib);
207  result ^= validate_payload(payload, m_mib);
208 
209  if (result.accepted()) {
210  using PendingPacket = PendingPacket<ShbPdu>;
211 
212  // step 4: set up packet repetition (NOTE 4 on page 57 requires re-execution of source operations)
213  if (request.repetition) {
214  // plaintext payload needs to get passed
215  m_repeater.add(request, *payload);
216  }
217 
218  // step 1: create PDU
219  auto pdu = create_shb_pdu(request);
220  pdu->common().payload = payload->size();
221 
222  ControlInfo ctrl(request);
223  auto transmit = [this, ctrl](PendingPacket::Packet&& packet) {
224  std::unique_ptr<ShbPdu> pdu;
225  std::unique_ptr<DownPacket> payload;
226  std::tie(pdu, payload) = std::move(packet);
227 
228  // update SO PV before actual transmission
229  pdu->extended().source_position = m_local_position_vector;
230 
231  // step 2: encapsulate packet by security
232  if (m_mib.itsGnSecurity) {
233  payload = encap_packet(ctrl.its_aid, *pdu, std::move(payload));
234  }
235 
236  // step 5: execute media-dependent procedures
237  execute_media_procedures(ctrl.communication_profile);
238 
239  // step 6: pass packet down to link layer with broadcast destination
240  pass_down(cBroadcastMacAddress, std::move(pdu), std::move(payload));
241 
242  // step 7: reset beacon timer
243  reset_beacon_timer();
244  };
245 
246  PendingPacket packet(std::make_tuple(std::move(pdu), std::move(payload)), transmit);
247 
248  // step 3: store & carry forwarding
249  if (request.traffic_class.store_carry_forward() && !m_location_table.has_neighbours()) {
250  PacketBuffer::data_ptr data { new PendingPacketBufferData<ShbPdu>(std::move(packet)) };
251  m_bc_forward_buffer.push(std::move(data), m_runtime.now());
252  } else {
253  // tranmsit immediately

255  }
256  }
257

261 DataConfirm Router::request(const GbcDataRequest& request, DownPacketPtr payload)
262 {
263  DataConfirm result;
264  result ^= validate_data_request(request, m_mib);
265  result ^= validate_payload(payload, m_mib);
266 
267  if (!result.accepted())
268  return result;
269 
270  // step 6: set up packet repetition
271  // packet repetition is done first because of "NOTE 2" on page 60:
272  // "For every retransmission, the source operations need to be re-executed".
273  // Hence, all routing decisions and security encapsulation have to be performed again.
274  // Assumption: "omit execution of further steps" does not cancel the repetition procedure.
275  if (request.repetition) {
276  m_repeater.add(request, *payload);
277  }
278 
279  using PendingPacket = PendingPacket<GbcPdu>;
280  using Packet = PendingPacket::Packet;
281 
282  // step 1: create PDU and set header fields
283  auto pdu = create_gbc_pdu(request);
284  pdu->common().payload = payload->size();
285 
286  ControlInfo ctrl(request);
287  auto transmit = [this, ctrl](Packet&& packet, const MacAddress& mac) {
288  std::unique_ptr<GbcPdu> pdu;
289  std::unique_ptr<DownPacket> payload;
290  std::tie(pdu, payload) = std::move(packet);
291 
292  // update SO PV before actual transmission
293  pdu->extended().source_position = m_local_position_vector;
294 
295  // step 5: apply security
296  if (m_mib.itsGnSecurity) {
297  assert(pdu->basic().next_header == NextHeaderBasic::Secured);
298  payload = encap_packet(ctrl.its_aid, *pdu, std::move(payload));
299  }
300 
301  // step 6: repetition is already set-up before
302 
303  // step 7: execute media-dependent procedures
304  execute_media_procedures(ctrl.communication_profile);
305 
306  // step 8: pass PDU to link layer
307  pass_down(mac, std::move(pdu), std::move(payload));
308  };
309 
310  auto forwarding = [this, transmit](Packet&& packet) {
311  // step 3: forwarding algorithm selection procedure
312  NextHop nh = forwarding_algorithm_selection(PendingPacketForwarding(std::move(packet), transmit), nullptr);
313 
314  // step 4: omit execution of further steps when packet if buffered or discarded
315  std::move(nh).process();
316  };
317 
318  PendingPacket packet(std::make_tuple(std::move(pdu), std::move(payload)), forwarding);
319 
320  // step 2: check if neighbours are present
321  const bool scf = request.traffic_class.store_carry_forward();
322  if (scf && !m_location_table.has_neighbours()) {
323  PacketBuffer::data_ptr data { new PendingPacketBufferData<GbcPdu>(std::move(packet)) };
324  m_bc_forward_buffer.push(std::move(data), m_runtime.now());
325  } else {
326  packet.process();
327  }
328 
329  return result;

497  switch (decap_confirm.report) {

499  case DecapReport::Invalid_Certificate:

505  case DecapReport::Signer_Certificate_Not_Found:
506  case DecapReport::Unsupported_Signer_Identifier_Type:
507  case DecapReport::Unencrypted_Message:

512  case DecapReport::Incompatible_Protocol:

588 NextHop Router::forwarding_algorithm_selection(PendingPacketForwarding&& packet, const LinkLayer* ll)
589 {
590  NextHop nh;
591  const Area& destination = packet.pdu().extended().destination(packet.pdu().common().header_type);
592  if (inside_or_at_border(destination, m_local_position_vector.position())) {
593  switch (m_mib.itsGnAreaForwardingAlgorithm) {
594  case BroadcastForwarding::Unspecified:
595  // do simple forwarding
596  case BroadcastForwarding::SIMPLE:
597  // Simple always returns link-layer broadcast address (see Annex F.2)
598  nh.transmit(std::move(packet), cBroadcastMacAddress);
599  break;
600  case BroadcastForwarding::CBF:
601  nh = area_contention_based_forwarding(std::move(packet), ll ? &ll->sender : nullptr);
602  break;
603  case BroadcastForwarding::Advanced:
604  nh = area_advanced_forwarding(std::move(packet), ll);
605  break;
606  default:
607  throw std::runtime_error("unhandled area forwarding algorithm");
608  break;
609  };
610  } else {
611  // packets received from senders located inside destination area are not meant for non-area forwarding
612  const LongPositionVector* pv_se = ll ? m_location_table.get_position(ll->sender) : nullptr;

750 NextHop Router::greedy_forwarding(PendingPacketForwarding&& packet)
751 {
752  NextHop nh;
753  GeodeticPosition dest = packet.pdu().extended().position();
754  const units::Length own = distance(dest, m_local_position_vector.position());
755  units::Length mfr_dist = own;
756 
757  MacAddress mfr_addr;
758  for (const LocationTableEntry& neighbour : m_location_table.neighbours()) {
759  if (neighbour.has_position_vector()) {
760  const units::Length dist = distance(dest, neighbour.get_position_vector().position());
761  if (dist < mfr_dist) {
762  mfr_addr = neighbour.link_layer_address();
763  mfr_dist = dist;
764  }
765  }
766  }
767 
768  if (mfr_dist < own) {
769  nh.transmit(std::move(packet), mfr_addr);
770  } else {
771  const bool scf = packet.pdu().common().traffic_class.store_carry_forward();
772  if (scf) {
773  std::function<void(PendingPacketForwarding&&)> greedy_fwd = [this](PendingPacketForwarding&& packet) {
774  NextHop nh = greedy_forwarding(std::move(packet));
775  std::move(nh).process();
776  };
777  PendingPacket<GbcPdu> greedy_packet(std::move(packet), greedy_fwd);
778  PacketBuffer::data_ptr data { new PendingPacketBufferData<GbcPdu>(std::move(greedy_packet)) };
779  m_bc_forward_buffer.push(std::move(data), m_runtime.now());
780  nh.buffer();
781  } else {
782  nh.transmit(std::move(packet), cBroadcastMacAddress);
783  }
784  }
785 
786  return nh;
787 }

789 NextHop Router::non_area_contention_based_forwarding(PendingPacketForwarding&& packet, const MacAddress* sender)
790 {
791  NextHop nh;
792  const GeoBroadcastHeader& gbc = packet.pdu().extended();
793  // immediately broadcast packet if it is originating from local router
794  if (!sender) {
795  nh.transmit(std::move(packet), cBroadcastMacAddress);

817  }

821  m_cbf_buffer.enqueue(std::move(cbf), clock_cast(to_cbf_max));

873 NextHop Router::area_advanced_forwarding(PendingPacketForwarding&& packet, const LinkLayer* ll)
874 {
875  NextHop nh;
876  if (!ll) {
877  // packet is from local node (source operations)
878  nh.transmit(std::move(packet), cBroadcastMacAddress);
879  } else {
880  const GeoBroadcastHeader& gbc = packet.pdu().extended();
881  const HeaderType ht = packet.pdu().common().header_type;
882  const Area destination_area = gbc.destination(ht);
883  const std::size_t max_counter = m_mib.vanetzaCbfMaxCounter;
884  auto cbf = m_cbf_buffer.fetch(gbc.source_position.gn_addr, gbc.sequence_number);
885 
886  if (cbf) {
887  // packet is already buffered
888  if (cbf->counter() >= max_counter) {
889  // stop contending if counter is exceeded
890  m_cbf_buffer.try_drop(gbc.source_position.gn_addr, gbc.sequence_number);
891  nh.discard();
892  } else if (!outside_sectorial_contention_area(cbf->sender(), ll->sender)) {
893  // within sectorial area
894  // - sender S = sender of buffered packet
895  // - forwarder F = sender of now received packet
896  m_cbf_buffer.try_drop(gbc.source_position.gn_addr, gbc.sequence_number);
897  nh.discard();
898  } else {
899  ++cbf->counter();
900  m_cbf_buffer.enqueue(std::move(*cbf), clock_cast(timeout_cbf(ll->sender)));
901  nh.buffer();
902  }
903  } else {
904  if (ll->destination == m_local_position_vector.gn_addr.mid()) {
905  // continue with greedy forwarding
906  nh = greedy_forwarding(packet.duplicate());
907  // optimization: avoid "double broadcast"
908  if (nh.valid() && nh.mac() == cBroadcastMacAddress) {
909  // contending without further broadcasting
910  static const PendingPacketForwarding::Function noop_fn =
911  [](PendingPacketForwarding::Packet&&, const MacAddress&) {};
912  PendingPacketForwarding noop { std::move(packet).packet(), noop_fn };
913  CbfPacket cbf { std::move(noop), ll->sender };
914  m_cbf_buffer.enqueue(std::move(cbf), clock_cast(m_mib.itsGnCbfMaxTime));
915  } else {
916  // no immediate broadcast by greedy forwarding
917  CbfPacket cbf { std::move(packet), ll->sender };
918  m_cbf_buffer.enqueue(std::move(cbf), clock_cast(m_mib.itsGnCbfMaxTime));
919  }
920  // next hop (nh) conveys result of greedy forwarding algorithm
921  } else {
922  // classical CBF (timeout_cbf_gbc looks up sender's position)
923  nh.buffer();
924  CbfPacket cbf { std::move(packet), ll->sender };
925  m_cbf_buffer.enqueue(std::move(cbf), clock_cast(timeout_cbf(ll->sender)));
926  }
927  }
928  }
929 
930  return nh;
931 }

933 bool Router::outside_sectorial_contention_area(const MacAddress& sender, const MacAddress& forwarder) const
934 {
935  using units::si::meter;
936  auto position_sender = m_location_table.get_position(sender);
937  auto position_forwarder = m_location_table.get_position(forwarder);
938 
939  // Assumption: if any position is missing, then sectorial area becomes infinite small
940  // As a result of this assumption, everything lays outside then
941  if (position_sender && position_forwarder) {
942  auto dist_r = distance(position_sender->position(), m_local_position_vector.position());
943  auto dist_f = distance(position_forwarder->position(), position_sender->position());
944  const auto dist_max = m_mib.itsGnDefaultMaxCommunicationRange;
945 
946  auto dist_rf = distance(position_forwarder->position(), m_local_position_vector.position());
947  auto angle_fsr = 0.0 * units::si::radians;
948  if (dist_r > 0.0 * meter && dist_f > 0.0 * meter) {
949  auto cos_fsr = (dist_rf * dist_rf - dist_r * dist_r - dist_f * dist_f) /
950  (-2.0 * dist_r * dist_f);
951  angle_fsr = boost::units::acos(cos_fsr);
952  }
953  const auto angle_th = m_mib.itsGnBroadcastCBFDefSectorAngle;
954 
955  return !(dist_r < dist_f && dist_f < dist_max && angle_fsr < angle_th);
956  } else {
957  return true;
958  }
959 }

1018 bool Router::process_extended(const ExtendedPduConstRefs<GeoBroadcastHeader>& pdu, const UpPacket& packet, const LinkLayer& ll)
1019 {
1020  // GBC forwarder and receiver operations (section 9.3.11.3 in EN 302 636-4-1 V1.2.1)
1021  const GeoBroadcastHeader& gbc = pdu.extended();
1022  const Address& source_addr = gbc.source_position.gn_addr;
1023  const Area dest_area = gbc.destination(pdu.common().header_type);
1024 
1025  // remember if LocTE(SO) exists (5) before duplicate packet detection might (3) silently create an entry
1026  const bool locte_exists = m_location_table.has_entry(source_addr);
1027 
1028  // step 3: determine position relative to destination area
1029  const bool within_destination = inside_or_at_border(dest_area, m_local_position_vector.position());
1030  // step 3a
1031  bool duplicate_packet = false;
1032  if (!within_destination) {
1033  if (m_mib.itsGnNonAreaForwardingAlgorithm == UnicastForwarding::Unspecified ||
1034  m_mib.itsGnNonAreaForwardingAlgorithm == UnicastForwarding::Greedy) {
1035  duplicate_packet = detect_duplicate_packet(source_addr, gbc.sequence_number);
1036  }
1037  // step 3b
1038  } else {
1039  if (m_mib.itsGnAreaForwardingAlgorithm == BroadcastForwarding::Unspecified ||
1040  m_mib.itsGnAreaForwardingAlgorithm == BroadcastForwarding::SIMPLE) {
1041  duplicate_packet = detect_duplicate_packet(source_addr, gbc.sequence_number);
1042  }
1043  }
1044  // step 3a & 3b
1045  if (duplicate_packet) {
1046  // omit execution of further steps
1047  return false;
1048  }
1049 
1050  // step 4: execute DAD
1051  detect_duplicate_address(source_addr, ll.sender);
1052 
1053  // step 5 & step 6 (make sure IS_NEIGHBOUR is false for new location table entry)
1054  const std::size_t packet_size = size(packet, OsiLayer::Network, OsiLayer::Application);
1055  auto& source_entry = m_location_table.update(gbc.source_position);
1056  source_entry.update_pdr(packet_size, m_mib.itsGnMaxPacketDataRateEmaBeta);
1057  if (!locte_exists) {
1058  // step 5b only
1059  source_entry.set_neighbour(false);
1060  }
1061 
1062  // step 7: pass packet to upper layer if router is within destination area, return value
1063 
1064  // step 8a: TODO: flush SO LS packet buffer if LS_pending, reset LS_pending
1065  // step 8b: flush UC forwarding packet buffer
1066  flush_unicast_forwarding_buffer(source_addr);
1067 
1068  // step 9: discard packet (no forwarding) if hop limit is reached
1069  if (pdu.basic().hop_limit <= 1) {
1070  forwarding_stopped(ForwardingStopReason::Hop_Limit);
1071  return within_destination; // discard packet (step 9a)
1072  } else if (m_mib.itsGnMaxPacketDataRate < std::numeric_limits<decltype(m_mib.itsGnMaxPacketDataRate)>::max()) {
1073  // do packet data rate checks (annex B.2) if set maximum rate is not "infinity" (i.e. max unsigned value)
1074  if (source_entry.get_pdr() > m_mib.itsGnMaxPacketDataRate * 1000.0) {
1075  forwarding_stopped(ForwardingStopReason::Source_PDR);
1076  return within_destination; // omit forwarding, source exceeds PDR limit
1077  } else if (const auto* sender_entry = m_location_table.get_entry(ll.sender)) {
1078  if (sender_entry->get_pdr() > m_mib.itsGnMaxPacketDataRate * 1000.0) {
1079  forwarding_stopped(ForwardingStopReason::Sender_PDR);
1080  return within_destination; // omit forwarding, sender exceeds PDR limit
1081  }
1082  }
1083  }
1084 
1085  // step 9b: update hop limit in basic header
1086  auto fwd_dup = create_forwarding_duplicate(pdu, packet);
1087  GbcPdu& fwd_pdu = get_pdu(fwd_dup);
1088  --fwd_pdu.basic().hop_limit;
1089  assert(fwd_pdu.basic().hop_limit + 1 == pdu.basic().hop_limit);
1090 
1091  using Packet = PendingPacketGbc::Packet;
1092 
1093  auto transmit = [this](Packet&& packet, const MacAddress& mac) {
1094  // step 13: execute media-dependent procedures
1095  execute_media_procedures(m_mib.itsGnIfType);
1096 
1097  // step 14: pass down to link-layer
1098  std::unique_ptr<Pdu> pdu;
1099  std::unique_ptr<DownPacket> payload;
1100  std::tie(pdu, payload) = std::move(packet);
1101 
1102  dcc::DataRequest request;
1103  request.destination = mac;
1104  request.source = m_local_position_vector.gn_addr.mid();
1105  request.dcc_profile = dcc::Profile::DP3;
1106  request.ether_type = geonet::ether_type;
1107  request.lifetime = std::chrono::seconds(pdu->basic().lifetime.decode() / units::si::seconds);
1108 
1109  pass_down(request, std::move(pdu), std::move(payload));
1110  };
1111 
1112  auto forwarding = [this, transmit, ll](Packet&& packet) {
1113  // step 11: execute forwarding algorithm
1114  PendingPacket<GbcPdu, const MacAddress&> tmp(std::move(packet), transmit);
1115  NextHop forwarding = forwarding_algorithm_selection(std::move(tmp), &ll);
1116 
1117  // step 12: transmit immediately if not buffered or discarded
1118  std::move(forwarding).process();
1119  };
1120 
1121  PendingPacketGbc fwd_packet(std::move(fwd_dup), forwarding);
1122 
1123  // step 10: store & carry forwarding procedure
1124  const bool scf = pdu.common().traffic_class.store_carry_forward();
1125  if (scf && !m_location_table.has_neighbours()) {
1126  PacketBuffer::data_ptr data { new PendingPacketBufferData<GbcPdu>(std::move(fwd_packet)) };
1127  m_bc_forward_buffer.push(std::move(data), m_runtime.now());
1128  } else {
1129  fwd_packet.process();
1130  }
1131

1133  return within_destination;
