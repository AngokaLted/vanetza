21 boost::optional<StartAndEndValidity> extract_validity_time(const Certificate& certificate)
22 {
23  boost::optional<StartAndEndValidity> restriction;
24 
25  for (auto& validity_restriction : certificate.validity_restriction) {
26  ValidityRestrictionType type = get_type(validity_restriction);
27 
28  if (type == ValidityRestrictionType::Time_Start_And_End) {
29  // reject more than one restriction
30  if (restriction) {
31  return boost::none;
32  }
33 
34  restriction = boost::get<StartAndEndValidity>(validity_restriction);
35 
36  // check if certificate validity restriction timestamps are logically correct
37  if (restriction->start_validity >= restriction->end_validity) {
38  return boost::none;
39  }
40  } else if (type == ValidityRestrictionType::Time_End) {
41  // must not be used, no certificate profile allows it
42  return boost::none;
43  } else if (type == ValidityRestrictionType::Time_Start_And_Duration) {
44  // must not be used, no certificate profile allows it
45  return boost::none;
46  }
47  }
48 
49  return restriction;
50 }

52 bool check_time_consistency(const Certificate& certificate, const Certificate& signer)
53 {
54  boost::optional<StartAndEndValidity> certificate_time = extract_validity_time(certificate);
55  boost::optional<StartAndEndValidity> signer_time = extract_validity_time(signer);
56 
57  if (!certificate_time || !signer_time) {
58  return false;
59  }
60 
61  if (signer_time->start_validity > certificate_time->start_validity) {
62  return false;
63  }
64 
65  if (signer_time->end_validity < certificate_time->end_validity) {
66  return false;
67  }
68 
69  return true;
70 }

96 bool check_permission_consistency(const Certificate& certificate, const Certificate& signer)
97 {
98  auto certificate_aids = extract_application_identifiers(certificate);
99  auto signer_aids = extract_application_identifiers(signer);
100  auto compare = [](ItsAid a, ItsAid b) { return a < b; };
101 
102  certificate_aids.sort(compare);
103  signer_aids.sort(compare);
104 
105  return std::includes(signer_aids.begin(), signer_aids.end(), certificate_aids.begin(), certificate_aids.end());
106 }

108 bool check_subject_assurance_consistency(const Certificate& certificate, const Certificate& signer)
109 {
110  auto certificate_assurance = certificate.get_attribute<SubjectAttributeType::Assurance_Level>();
111  auto signer_assurance = signer.get_attribute<SubjectAttributeType::Assurance_Level>();
112 
113  if (!certificate_assurance || !signer_assurance) {
114  return false;
115  }
116 
117  // See TS 103 096-2 v1.3.1, section 5.2.7.11 + 5.3.5.17 and following
118  if (certificate_assurance->assurance() > signer_assurance->assurance()) {
119  return false;
120  } else if (certificate_assurance->assurance() == signer_assurance->assurance()) {
121  if (certificate_assurance->confidence() > signer_assurance->confidence()) {
122  return false;
123  }
124  }
125 
126  return true;
127 }

175 CertificateValidity DefaultCertificateValidator::check_certificate(const Certificate& certificate)
176 {
177  if (!extract_validity_time(certificate)) {
178  return CertificateInvalidReason::Broken_Time_Period;
179  }
180 
181  if (!certificate.get_attribute<SubjectAttributeType::Assurance_Level>()) {
182  return CertificateInvalidReason::Missing_Subject_Assurance;
183  }
184 
185  SubjectType subject_type = certificate.subject_info.subject_type;
186 
187  // check if subject_name is empty if certificate is authorization ticket
188  if (subject_type == SubjectType::Authorization_Ticket && 0 != certificate.subject_info.subject_name.size()) {
189  return CertificateInvalidReason::Invalid_Name;
190  }
191 
192  if (get_type(certificate.signer_info) != SignerInfoType::Certificate_Digest_With_SHA256) {
193  return CertificateInvalidReason::Invalid_Signer;
194  }
195 
196  HashedId8 signer_hash = boost::get<HashedId8>(certificate.signer_info);
197 
198  // try to extract ECDSA signature
199  boost::optional<EcdsaSignature> sig = extract_ecdsa_signature(certificate.signature);
200  if (!sig) {
201  return CertificateInvalidReason::Missing_Signature;
202  }
203 
204  // create buffer of certificate
205  ByteBuffer binary_cert = convert_for_signing(certificate);
206 
207  // authorization tickets may only be signed by authorization authorities
208  if (subject_type == SubjectType::Authorization_Ticket) {
209  for (auto& possible_signer : m_cert_cache.lookup(signer_hash, SubjectType::Authorization_Authority)) {
210  auto verification_key = get_public_key(possible_signer, m_crypto_backend);
211  if (!verification_key) {
212  continue;
213  }
214 
215  if (m_crypto_backend.verify_data(verification_key.get(), binary_cert, sig.get())) {
216  if (!check_consistency(certificate, possible_signer)) {
217  return CertificateInvalidReason::Inconsistent_With_Signer;
218  }
219 
220  return CertificateValidity::valid();
221  }
222  }
223  }
224 
225  // authorization authorities may only be signed by root CAs
226  // Note: There's no clear specification about this, but there's a test for it in 5.2.7.12.4 of TS 103 096-2 V1.3.1
227  if (subject_type == SubjectType::Authorization_Authority) {
228  for (auto& possible_signer : m_trust_store.lookup(signer_hash)) {
229  auto verification_key = get_public_key(possible_signer, m_crypto_backend);
230  if (!verification_key) {
231  continue;
232  }
233 
234  if (m_crypto_backend.verify_data(verification_key.get(), binary_cert, sig.get())) {
235  if (!check_consistency(certificate, possible_signer)) {
236  return CertificateInvalidReason::Inconsistent_With_Signer;
237  }
238 
239  return CertificateValidity::valid();
240  }
241  }
242  }
243 
244  return CertificateInvalidReason::Unknown_Signer;
245 }
