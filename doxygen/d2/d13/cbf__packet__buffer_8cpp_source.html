73 bool CbfPacketBuffer::try_drop(const Address& src, SequenceNumber sn)
74 {
75  bool packet_dropped = false;
76 
77  auto& id_map = m_timers.right;
78  auto found = id_map.find(identifier(src, sn));
79  if (found != id_map.end()) {
80  auto& packet = found->info;
81  m_stored_bytes -= packet->length();
82  m_packets.erase(packet);
83  auto& timer_map = m_timers.left;
84  auto successor = timer_map.erase(m_timers.project_left(found));
85  if (successor == timer_map.begin() && !timer_map.empty()) {
86  // erased timer was scheduled one, reschedule timer trigger
87  schedule_timer();
88  }
89  packet_dropped = true;
90  }
91 
92  assert(m_packets.size() == m_timers.size());
93  return packet_dropped;

96 void CbfPacketBuffer::enqueue(CbfPacket&& packet, Clock::duration timeout)
97 {
98  if(timeout <= Clock::duration::zero()) return;
99  m_stored_bytes += packet.length();
100 
101  // do head drop if necessary
102  while (m_stored_bytes > m_capacity_bytes && !m_packets.empty()) {
103  m_stored_bytes -= m_packets.front().length();
104  m_timers.right.erase(identifier(m_packets.front()));
105  m_packets.pop_front();
106  }
107 
108  Timer timer = { m_runtime, timeout };
109  Identifier id = identifier(packet);
110  m_packets.emplace_back(std::move(packet));
111  using timer_value = timer_bimap::value_type;
112  auto insertion = m_timers.insert(timer_value { timer, id, std::prev(m_packets.end()) });
113  if (!insertion.second) {
114  m_stored_bytes -= m_packets.back().length();
115  m_packets.pop_back();
116  assert(m_packets.size() == m_timers.size());
117  throw std::runtime_error("Illegal insertion of duplicate CBF packet");
118  } else if (m_timers.project_left(insertion.first) == m_timers.left.begin()) {
119  // enqueued packet expires first, reschedule timer trigger
120  schedule_timer();
121  }
122  assert(m_packets.size() == m_timers.size());
123 }

157 void CbfPacketBuffer::flush()
158 {
159  // fetch all expired timers
160  const Timer now { m_runtime, std::chrono::seconds(0) };
161  auto end = m_timers.left.upper_bound(now);
162  for (auto it = m_timers.left.begin(); it != end;) {
163  // reduce LT by queuing time
164  const Timer& timer = it->first;
165  CbfPacket& packet = *it->info;
166  bool valid_packet = reduce_lifetime(timer, packet);
167  m_stored_bytes -= packet.length();
168  if (valid_packet) {
169  m_timer_callback(std::move(packet).packet());
170  }
171 
172  m_packets.erase(it->info);
173  it = m_timers.left.erase(it);
174  }
175 
176  // schedule timer if not empty
177  if (!m_timers.empty()) {
178  schedule_timer();
179  }
180 }
