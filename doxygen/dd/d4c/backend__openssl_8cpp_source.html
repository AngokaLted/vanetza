27 {
28  auto priv_key = internal_private_key(key);
29  auto digest = calculate_digest(data);
30 
31  // sign message data represented by the digest
32  openssl::Signature signature { ECDSA_do_sign(digest.data(), digest.size(), priv_key) };
33 #if OPENSSL_API_COMPAT < 0x10100000L
34  const BIGNUM* sig_r = signature->r;
35  const BIGNUM* sig_s = signature->s;
36 #else
37  const BIGNUM* sig_r = nullptr;
38  const BIGNUM* sig_s = nullptr;
39  ECDSA_SIG_get0(signature, &sig_r, &sig_s);
40 #endif
41 
42  EcdsaSignature ecdsa_signature;
43  X_Coordinate_Only coordinate;
44 
45  if (sig_r && sig_s) {
46  const size_t len = field_size(PublicKeyAlgorithm::ECDSA_NISTP256_With_SHA256);
47 
48  const auto num_bytes_s = BN_num_bytes(sig_s);
49  assert(len >= static_cast<size_t>(num_bytes_s));
50  ecdsa_signature.s.resize(len, 0x00);
51  BN_bn2bin(sig_s, ecdsa_signature.s.data() + len - num_bytes_s);
52 
53  const auto num_bytes_r = BN_num_bytes(sig_r);
54  assert(len >= static_cast<size_t>(num_bytes_r));
55  coordinate.x.resize(len, 0x00);
56  BN_bn2bin(sig_r, coordinate.x.data() + len - num_bytes_r);
57  } else {
58  throw openssl::Exception();
59  }
60 
61  ecdsa_signature.R = std::move(coordinate);
62  return ecdsa_signature;

75 {
76  struct DecompressionVisitor : public boost::static_visitor<bool>
77  {
78  bool operator()(const X_Coordinate_Only&)
79  {
80  return false;
81  }
82

84  {

86  }

88  bool operator()(const Compressed_Lsb_Y_1& p)
89  {
90  return decompress(p.x, 1);
91  }
92 
93  bool operator()(const Uncompressed& p)
94  {
95  result = p;
96  return true;
97  }

99  bool decompress(const ByteBuffer& x, int y_bit)
100  {
101  openssl::BigNumberContext ctx;
102  openssl::BigNumber x_coordinate(x);
103  openssl::Group group(NID_X9_62_prime256v1);
104  openssl::Point point(group);
105  openssl::BigNumber y_coordinate;
106 
107  result.x = x;
108  result.y.resize(result.x.size());
109 
110 #if OPENSSL_API_COMPAT < 0x10101000L
111  EC_POINT_set_compressed_coordinates_GFp(group, point, x_coordinate, y_bit, ctx);
112  EC_POINT_get_affine_coordinates_GFp(group, point, nullptr, y_coordinate, ctx);
113  std::size_t y_coordinate_bytes = BN_num_bytes(y_coordinate);
114  if (y_coordinate_bytes <= result.y.size()) {
115  BN_bn2bin(y_coordinate, result.y.data() + (result.y.size() - y_coordinate_bytes));
116  return true;
117  } else {
118  return false;
119  }
120 #else
121  EC_POINT_set_compressed_coordinates(group, point, x_coordinate, y_bit, ctx);
122  EC_POINT_get_affine_coordinates(group, point, nullptr, y_coordinate, ctx);
123  return (BN_bn2binpad(y_coordinate, result.y.data(), result.y.size()) != -1);
124 #endif
125  }
126 
127  Uncompressed result;
128  };
129 
130  DecompressionVisitor visitor;
131  if (boost::apply_visitor(visitor, ecc_point)) {
132  return visitor.result;
133  } else {
134  return boost::none;
135  }
136 }

141

150 openssl::Key BackendOpenSsl::internal_private_key(const ecdsa256::PrivateKey& generic) const
151 {
152  openssl::Key key(NID_X9_62_prime256v1);
153  openssl::BigNumber prv(generic.key);
154  EC_KEY_set_private_key(key, prv);
155 
156  // OpenSSL requires public key, so we recreate it from private key
157  openssl::BigNumberContext ctx;
158  const EC_GROUP* group = EC_KEY_get0_group(key);
159  openssl::Point pub(group);
160  openssl::check(EC_POINT_mul(group, pub, prv, nullptr, nullptr, ctx));
161  EC_KEY_set_public_key(key, pub);
162 
163  openssl::check(EC_KEY_check_key(key));
164  return key;
165 }
