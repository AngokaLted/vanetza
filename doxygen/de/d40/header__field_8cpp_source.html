178 std::size_t deserialize(InputArchive& ar, std::list<HeaderField>& list)
179 {
180  const std::size_t size = trim_size(deserialize_length(ar));
181  std::size_t read = 0;
182  boost::optional<SymmetricAlgorithm> sym_algo;
183  while (read < size) {
184  HeaderField field;
185  HeaderFieldType type;
186  deserialize(ar, type);
187  read += sizeof(HeaderFieldType);
188  switch (type) {
189  case HeaderFieldType::Generation_Time: {
190  Time64 time;
191  deserialize(ar, time);
192  field = time;
193  list.push_back(field);
194  read += sizeof(Time64);
195  break;
196  }
197  case HeaderFieldType::Generation_Time_Confidence: {
198  Time64WithStandardDeviation time;
199  deserialize(ar, time.time64);
200  deserialize(ar, time.log_std_dev);
201  field = time;
202  list.push_back(field);
203  read += sizeof(Time64);
204  read += sizeof(uint8_t);
205  break;
206  }
207  case HeaderFieldType::Expiration: {
208  Time32 time;
209  deserialize(ar, time);
210  field = time;
211  list.push_back(field);
212  read += sizeof(Time32);
213  break;
214  }
215  case HeaderFieldType::Generation_Location: {
216  ThreeDLocation loc;
217  read += deserialize(ar, loc);
218  field = loc;
219  list.push_back(field);
220  break;
221  }
222  case HeaderFieldType::Request_Unrecognized_Certificate: {
223  const std::size_t tmp_size = trim_size(deserialize_length(ar));
224  read += tmp_size;
225  std::list<HashedId3> hashedList;
226  for (std::size_t c = 0; c < tmp_size; c += 3) {
227  HashedId3 id;
228  ar >> id[0];
229  ar >> id[1];
230  ar >> id[2];
231  hashedList.push_back(id);
232  }
233  field = hashedList;
234  read += length_coding_size(tmp_size);
235  list.push_back(field);
236  break;
237  }
238  case HeaderFieldType::Its_Aid: {
239  IntX its_aid;
240  read += deserialize(ar, its_aid);
241  field = its_aid;
242  list.push_back(field);
243  break;
244  }
245  case HeaderFieldType::Signer_Info: {
246  SignerInfo info;
247  read += deserialize(ar, info);
248  field = info;
249  list.push_back(field);
250  break;
251  }
252  case HeaderFieldType::Recipient_Info: {
253  std::list<RecipientInfo> recipientList;
254  if (sym_algo) {
255  const size_t tmp_size = deserialize(ar, recipientList, sym_algo.get());
256  read += tmp_size;
257  read += length_coding_size(tmp_size);
258  field = recipientList;
259  list.push_back(field);
260  } else {
261  throw deserialization_error("HeaderFields: RecipientInfo read before EncryptionParameters: SymmetricAlgorithm still unknown");
262  }
263  break;
264  }
265  case HeaderFieldType::Encryption_Parameters: {
266  EncryptionParameter param;
267  read += deserialize(ar, param);
268  field = param;
269  sym_algo = get_type(param);
270  list.push_back(field);
271  break;
272  }
273  default:
274  throw deserialization_error("Unknown HeaderFieldType");
275  break;
276  }
277  }
278  return size;
279 }
