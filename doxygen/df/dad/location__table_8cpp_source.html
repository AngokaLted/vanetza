33 void LocationTableEntry::update_pdr(std::size_t packet_size, double beta)
34 {
35  using namespace vanetza::units;
36 
37  if (std::isnan(m_pdr)) {
38  m_pdr = 0.0;
39  m_pdr_update = m_runtime.now();
40  } else if (beta > 0.0 && beta < 1.0) {
41  const std::chrono::duration<double> time_period = m_runtime.now() - m_pdr_update;
42  if (time_period.count() > 0.0) {
43  double instant_pdr = packet_size / time_period.count();
44  m_pdr *= beta;
45  m_pdr += (1.0 - beta) * instant_pdr;
46  m_pdr_update = m_runtime.now();
47  }
