21 {
22  // renew certificate if necessary
23  for (auto& validity_restriction : m_own_certificate.validity_restriction) {

26  if (start_and_end && start_and_end->end_validity < renewal_deadline) {

77 Certificate NaiveCertificateProvider::generate_authorization_ticket()
78 {
79  // create certificate
80  Certificate certificate;
81 
82  // section 6.1 in TS 103 097 v1.2.1
83  certificate.signer_info = calculate_hash(aa_certificate());
84 
85  // section 6.3 in TS 103 097 v1.2.1
86  certificate.subject_info.subject_type = SubjectType::Authorization_Ticket;
87  // section 7.4.2 in TS 103 097 v1.2.1, subject_name implicit empty
88 
89  // set assurance level
90  certificate.subject_attributes.push_back(SubjectAssurance(0x00));
91 
92  certificate.add_permission(aid::CA, ByteBuffer({ 1, 0, 0 }));
93  certificate.add_permission(aid::GN_MGMT, ByteBuffer({})); // required for beacons
94  certificate.add_permission(aid::IPV6_ROUTING, ByteBuffer({})); // required for routing tests
95 
96  // section 7.4.1 in TS 103 097 v1.2.1
97  // set subject attributes
98  // set the verification_key
99  Uncompressed coordinates;
100  coordinates.x.assign(m_own_key_pair.public_key.x.begin(), m_own_key_pair.public_key.x.end());
101  coordinates.y.assign(m_own_key_pair.public_key.y.begin(), m_own_key_pair.public_key.y.end());
102  EccPoint ecc_point = coordinates;
103  ecdsa_nistp256_with_sha256 ecdsa;
104  ecdsa.public_key = ecc_point;
105  VerificationKey verification_key;
106  verification_key.key = ecdsa;
107  certificate.subject_attributes.push_back(verification_key);
108 
109  // section 6.7 in TS 103 097 v1.2.1
110  // set validity restriction
111  StartAndEndValidity start_and_end;
112  start_and_end.start_validity = convert_time32(m_runtime.now() - std::chrono::hours(1));
113  start_and_end.end_validity = convert_time32(m_runtime.now() + std::chrono::hours(23));
114  certificate.validity_restriction.push_back(start_and_end);
115 
116  sign_authorization_ticket(certificate);
117 
118  return certificate;
119 }
