91  {
92  ByteBuffer compact;
93  compact.reserve(x.size() + 1);
94  compact.push_back(type);
95  std::copy(x.begin(), x.end(), std::back_inserter(compact));
96 
97  BackendCryptoPP::Point point;
98  CryptoPP::DL_GroupParameters_EC<CryptoPP::ECP> group(CryptoPP::ASN1::secp256r1());
99  group.GetCurve().DecodePoint(point, compact.data(), compact.size());
100 
101  result.x = x;
102  result.y.resize(result.x.size());
103  point.y.Encode(result.y.data(), result.y.size());
104  }
105 
106  Uncompressed result;
107  };
